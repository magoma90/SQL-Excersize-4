SELECT STUDENT NAME,TEST_SCORE
CASE
WHEN TEST_SCORE BETWEEN 90 AND 100 THEN 'EXCELLENT'
WHEN TEST_SCORE BETWEEN 75 AND 89 THEN ' GOOD'
WHEN TEST_SCORE BETWEEN 50 AND 74 THEN 'AVERAGE'
WHEN TEST_SCORE< 50 THEN 'POOR'
ELSE 'FAIL'
END AS PERFORMANCE
FROM STUDENT;

SELECT SALES_CATEGORY,TOTAL_AMOUNT
CASE
WHEN TOTAL_AMOUNT BETWEEN 95,00 AND 35,00 THEN 'LOW'
WHEN TOTAL_AMOUNT BETWEEN 480,00 THEN 'MEDIUM'
WHEN TOTAL_AMOUNT> 1200,00 THEN 'HIGH'

SELECT BULK SALE,REGULAR SALE
CASE
WHEN SALE BETWEEN 1AND 3 THEN 'REGULAR'
WHEN SALE >=15 THEN 'BULKSALE'
END
FROM SALES
